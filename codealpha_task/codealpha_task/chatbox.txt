import nltk
import random
import string

nltk.download('punkt')
nltk.download('wordnet')

from nltk.chat.util import reflections
from nltk.stem import WordNetLemmatizer

responses = {
    'hello': ['Hi there!', 'Hello!', 'Hey! How can I help you?'],
    'how are you': ['I am a bot, but I am doing great!', 'All systems go!', 'Feeling chatty today!'],
    'what is your name': ['You can call me ChatBot.', 'I am your virtual assistant.'],
    'bye': ['Goodbye!', 'See you later!', 'Bye! Have a great day!']
}

lemmatizer = WordNetLemmatizer()

def preprocess(text):
    text = text.lower()
    tokens = nltk.word_tokenize(text)
    tokens = [lemmatizer.lemmatize(word) for word in tokens if word not in string.punctuation]
    return tokens

def get_response(user_input):
    tokens = preprocess(user_input)
    joined = ' '.join(tokens)

    for key in responses:
        if key in joined:
            return random.choice(responses[key])

    return "I'm not sure I understand. Can you rephrase that?"

def chat():
    print("ðŸ¤– ChatBot: Hello! Type 'bye' to exit.")
    while True:
        user_input = input("You: ")
        if user_input.lower() in ['bye', 'exit', 'quit']:
            print("ðŸ¤– ChatBot:", random.choice(responses['bye']))
            break
        response = get_response(user_input)
        print("ðŸ¤– ChatBot:", response)

if __name__ == "__main__":
    chat()
