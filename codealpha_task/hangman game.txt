import random

word_list = ['python', 'hangman', 'programming', 'challenge', 'developer']

chosen_word = random.choice(word_list)
word_length = len(chosen_word)

display = ['_' for _ in range(word_length)]
guessed_letters = []
lives = 6  # Number of incorrect guesses allowed

print("ğŸ® Welcome to Hangman!")
print("Guess the word, one letter at a time.")
print("You have", lives, "lives.\n")

while lives > 0 and '_' in display:
    print("Current word:", ' '.join(display))
    guess = input("Guess a letter: ").lower()

    if not guess.isalpha() or len(guess) != 1:
        print("Please enter a single alphabet letter.\n")
        continue

    if guess in guessed_letters:
        print(f"You already guessed '{guess}'. Try a different letter.\n")
        continue

    guessed_letters.append(guess)

    if guess in chosen_word:
        for i in range(word_length):
            if chosen_word[i] == guess:
                display[i] = guess
        print(f"âœ… Good guess!\n")
    else:
        lives -= 1
        print(f"âŒ Wrong guess! You have {lives} lives left.\n")

if '_' not in display:
    print("ğŸ‰ Congratulations! You guessed the word:", chosen_word)
else:
    print("ğŸ’€ Game Over! The word was:", chosen_word)
